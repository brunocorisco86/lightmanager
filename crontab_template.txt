# Crontab para Light Manager
# Diretorio do Projeto: /home/bruno/lightmanager

# --- MANUTENÇÃO DO SISTEMA ---

# 0. Garante permissões do Mosquitto no Boot
@reboot sudo chown mosquitto:mosquitto /etc/mosquitto/passwd && sudo chmod 600 /etc/mosquitto/passwd && sudo mkdir -p /var/log/mosquitto && sudo chown mosquitto:mosquitto /var/log/mosquitto

# 1. Watchdog: Garante que o Mosquitto esteja sempre de pé (a cada 5 min)
*/5 * * * * pgrep -x mosquitto > /dev/null || sudo rc-service mosquitto start

# --- PROCESSOS PERSISTENTES (REBOOT COM DELAY) ---

# Usa /bin/bash explicitamente para evitar problemas com o shell padrao do cron
@reboot sleep 40 && /bin/bash /home/bruno/lightmanager/scripts/entrypoint.sh api
@reboot sleep 50 && /bin/bash /home/bruno/lightmanager/scripts/entrypoint.sh solar
@reboot sleep 60 && /bin/bash /home/bruno/lightmanager/scripts/entrypoint.sh bot

# --- TAREFAS AGENDADAS ---

# 4. Gerar relatorio diario de consumo (Todo dia as 23:55)
55 23 * * * /bin/bash /home/bruno/lightmanager/reports/generate_daily.sh >> /home/bruno/lightmanager/logs/cron.log 2>&1

# 5. Backup semanal para Cloudflare R2 (Todo domingo as 03:00 da manha)
00 03 * * 0 /bin/bash /home/bruno/lightmanager/scripts/backup_r2.sh >> /home/bruno/lightmanager/logs/backup.log 2>&1
